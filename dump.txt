function displayDataScreen() {
  // Read the user-info.json file and parse the JSON data
  let userData = JSON.parse(fs.readFileSync("user-info.json"));

  // Print the entire object using console.dir()
  console.dir(userData, { depth: null });

  // Prompt the user for the key of the object they want to delete
  const key = prompt("Enter the key of the object you want to delete: ");

  // Check if the key exists in the object
  if (key in userData) {
    // Delete the object from the object
    delete userData[key];
  } else {
    console.log("Invalid key.");
  }

  // Write the updated object back to the JSON file
  fs.writeFileSync("user-info.json", JSON.stringify(userData));
}
